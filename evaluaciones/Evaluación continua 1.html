Texto de la respuesta

function [T] = jacobi_T(A)

  D_s = diag(diag(A));

  L_s = tril(A,-1);

  U_s = triu(A,1);

  T = - inv(D_s)*(L_s+U_s);

endfunction


function [T] = gauss_seidel_T(A)

  D_s = diag(diag(A));

  L_s = tril(A,-1);

  U_s = triu(A,1);

  T = - inv(D_s+L_s)*U_s;

endfunction


function [T] = sor_T(A,w)

  D_s = diag(diag(A));

  L_s = tril(A,-1);

  U_s = triu(A,1);

  T = - inv(D_s+w*L_s) * ((1-w)*D_s-w*U_s);

endfunction
